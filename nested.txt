package train;

import java.util.*;

public class nested {
	public static void main(String args[]) {

		
		 for(int i=0;i<5;i++) { for(int j=0;j<i;j++) { System.out.print("*"); }
		  System.out.println();}
		

		
		 for(int i=5;i>0;i--) { for(int j=0;j<i;j++) { System.out.print("*"); }
		 System.out.println();
		 

		for (int k = 5; k > 0; k--) {
			for (int j = k;j<5; j--) {
				System.out.print("*");
			}
			System.out.println();

		}
	}
	}
}
